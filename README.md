# nzpure


# fast start

      - clone into theme/
            git clone git@github.com:NadirZenith/nzpure.git

      - install node dps
            npm install

      - compile assets
            development - grunt dev
            production - grunt build
            
      - wp-config settings
            development - define( 'WP_ENV', 'development' );
            production - default
